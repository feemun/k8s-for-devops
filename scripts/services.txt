# 格式：服务名 命名空间 宿主机监听端口(NodePort) Pod内容器端口(targetPort)
# 示例：redis-nodeport redis-service 30379 6379
# 访问方式：
# - 宿主机本机：redis-cli -h 127.0.0.1 -p 30379
# - 局域网其他机器：redis-cli -h <宿主机IP> -p 30379
# 示例：宿主机IP是 192.168.1.100，则局域网访问：redis-cli -h 192.168.1.100 -p 30379

mysql-nodeport        mysql           30306  3306
redis-nodeport        redis-service   30379  6379
elasticsearch-nodeport elasticsearch   30200  9200
